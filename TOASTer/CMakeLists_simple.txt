cmake_minimum_required(VERSION 3.15)

project(TOASTerSimple VERSION 1.0.0)

set(CMAKE_CXX_STANDARD 17)

# Just try to build with minimal JUCE setup
include(FetchContent)
FetchContent_Declare(
    JUCE
    GIT_REPOSITORY https://github.com/juce-framework/JUCE.git
    GIT_TAG 8.0.4
)
FetchContent_MakeAvailable(JUCE)

# Create minimal app
juce_add_gui_app(TOASTerSimple
    PRODUCT_NAME "TOASTerSimple"
    COMPANY_NAME "JAMNet"
    BUNDLE_ID "com.jamnet.toastersimple"
    VERSION "1.0.0"
)

# Add minimal source
target_sources(TOASTerSimple PRIVATE
    Source/Main.cpp
    Source/MainComponent.cpp
)

# Minimal JUCE modules
target_link_libraries(TOASTerSimple PRIVATE
    juce::juce_gui_basics
    juce::juce_core
)

# Disable code signing
set_target_properties(TOASTerSimple PROPERTIES
    XCODE_ATTRIBUTE_CODE_SIGN_IDENTITY ""
    XCODE_ATTRIBUTE_CODE_SIGNING_REQUIRED NO
)

target_compile_definitions(TOASTerSimple PRIVATE
    JUCE_WEB_BROWSER=0
    JUCE_USE_CURL=0
    JUCE_DISPLAY_SPLASH_SCREEN=0
)
